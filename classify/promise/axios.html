Http相关 #

常见状态码

  1. 200：请求成功
  2. 201：已创建
  3. 401：未授权/请求用户身份认证
  4. 404：服务器无法根据请求找到资源
  5. 500：服务器内部错误，无法完成请求

json-server #

快速构建一台服务器

bash
# 全局安装json-server
$ npm i json-server -g
$ json-server db.json

axios #

特点

  1. 基于promise的异步ajax请求库
  2. 浏览器和node端都可以使用
  3. 支持请求/响应拦截器
  4. 支持请求取消
  5. 请求/响应数据转换
  6. 批量发送多个请求

get请求

@from:server.js

js
const express = require('express')
const cors = require('cors')
const app = express()

// 注册中间件
app.use(cors())
// 配置解析JSON格式中间件
app.use(express.urlencoded({ extended: true }))
app.use(express.json())

//路由
app.get('/posts', (req, res) => {
    console.log(req.query); // { title: 'TS学习' }
    res.send('响应成功')
})

app.post('/comments', (req, res) => {
    console.log(req.body); // { body: '评论1' }
    res.send('响应成功')
})

// 服务器启动
app.listen(80, () => {
    console.log('服务器启动在http://127.0.0.1');
})

拦截器 #

请求拦截器

@method:axios.interceptors.request.use(config => ...，err => ...)

@description:对所有请求做统一处理 => 追加请求头、追加参数、界面loading提示

响应拦截器

@method:axios.interceptors.request.use(response => ...,err => ...)

@description:若请求成功，对成功的数据进行处理；若失败，对失败进行统一处理

中断请求 #

取消请求

js
const { CancelToken, isCancel } = axios // 解构出CancelToken构造函数
let cancel // 定义全局cancel
btn.onclick = () => {
    if (cancel) cancel() // 取消用户发送的多次请求
    axios({
        // 取消请求令牌
        cancelToken: new CancelToken((c) => {
            cancel = c
        }),
		...
    }).then(res => ...).catch(err => {
        // 进入判断说明是用户取消的请求
        if(isCancel(err)){
            console.log('用户取消了请求，原因是:',err.message)
        } else {
            ...
        }
    })
}
cancelbtn.onclick = () => {
    cancel('用户多次点击发送请求') // 调用CancelToken构造函数的执行回调函数，则可以取消本次请求
}

配合请求响应拦截器使用

js
// @ts-nocheck
const btn = document.querySelector('.button')
const cancelbtn = document.querySelector('.button2')
// 解构获取取消令牌的构造函数、判断是否为用户导致取消的请求函数
const { CancelToken, isCancel } = axios

let cancel
// 请求拦截器(对请求做统一处理)
axios.interceptors.request.use(config => {
    if (cancel) cancel('用户多次点击发送请求') // 取消用户发送的上一次相同请求
    // 向请求头追加取消响应令牌
    config.cancelToken = new CancelToken((c) => {
        cancel = c
    })
    return config
})

// 响应拦截器
axios.interceptors.response.use(res => {
    return res.data
}, err => {
    if (isCancel(err)) {
        console.log('请求取消:', err.message);
    } else {
        console.log(err);
    }
    // 中断promise
    return new Promise(() => { })
})


btn.onclick = async () => {
    let result = await axios.post('http://127.0.0.1/comments', {
        body: '评论2'
    })
    console.log(result);
}

cancelbtn.onclick = () => {
    cancel('用户主动取消请求') // 调用CancelToken构造函数的执行回调函数，则可以取消本次请求
}

批量发送请求 #

axios.all([...]).then(...).catch(...)

js
const btn = document.querySelector('.button')

btn.onclick = () => {
    // axios内置的API，类似于Promise.all()
    axios.all([
        axios.get('http://127.0.0.1/posts', {
            params: { title: 'TS学习' }
        }),
        axios.post('http://127.0.0.1/comments',{
            body: '评论1' 
        })
    ]).then((result) => {
       // 处理返回数据，只拿响应数据
       let res = result.reduce((pre,cur) => {
        return pre.concat(cur.data)
       },[])
       console.log(res);
    }).catch((err) => {
        console.log(err);
    });
}
